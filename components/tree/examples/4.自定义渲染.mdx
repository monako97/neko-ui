通过数据渲染

```jsx
function Demo() {
  const data = [
    {
      title: '文件名称',
      subTitle: 'object',
      key: 'a',
      children: [
        {
          title: '备注',
          subTitle: 'object',
          key: 'a.b',
          children: [
            {
              title: '备注',
              subTitle: 'string',
              key: 'a.b.x',
            },
          ],
        },
      ],
    },
    {
      title: '编号',
      subTitle: 'object',
      key: 'b',
      children: [
        {
          title: '备注',
          subTitle: 'string',
          key: 'b2',
        },
        {
          title: '是否有效',
          subTitle: 'boolean',
          key: 'c2',
        },
      ],
    },
  ];

  return (
    <Tree
      data={data}
      renderRow={(item, title, subTitle) => [
        <i key={item.key} style={{ color: 'pink' }}>{title}</i>,
        <b key={item.key + item.title}>{subTitle}</b>,
        <i key="x" style={{ flex: 1, textAlign: 'right' }}>x</i>
      ]}
    />
  );
}
```
