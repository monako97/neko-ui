Tree 内置有大、中、小三种尺寸。
通过设置 `size` 为 `large`、`small` 分别把按钮设为大、小尺寸。若不设置 `size`，则尺寸为中 `normal`。

```jsx
function Demo() {
  const [size, setSize] = React.useState('normal');
  const opt = [
    { label: '小', value: 'small' },
    { label: '默认', value: 'normal' },
    { label: '大', value: 'large' },
  ];
  const data = [
    {
      title: '文件名称',
      subTitle: 'object',
      key: 'a',
      children: [
        {
          title: '备注',
          subTitle: 'object',
          key: 'a.b',
          children: [
            {
              title: '备注',
              subTitle: 'string',
              key: 'a.b.x',
            },
          ],
        },
      ],
    },
    {
      title: '编号',
      subTitle: 'object',
      key: 'b',
      children: [
        {
          title: '备注',
          subTitle: 'string',
          key: 'b2',
        },
        {
          title: '是否有效',
          subTitle: 'boolean',
          key: 'c2',
        },
      ],
    },
  ];

  return (
    <>
      {opt.map((item) => {
        return (
          <Button
            type="primary"
            fill={item.value === size}
            key={item.value}
            onClick={() => setSize(item.value)}
          >
            {item.label}
          </Button>
        );
      })}
      <Tree data={data} size={size} />
    </>
  );
}
```
