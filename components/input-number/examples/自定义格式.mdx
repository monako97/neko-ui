function Demo() {
  const [val, setVal] = React.useState(0);

  return (
    <InputNumber
      min={0}
      max={1}
      step={0.01}
      formatter={(v) => `${(v || 0) * 100}%`}
      parser={(v) => parseFloat(v?.toString().replace(/%$/, '') || '0') / 100}
      value={val}
      onChange={setVal}
      suffix={val}
    />
  );
}
