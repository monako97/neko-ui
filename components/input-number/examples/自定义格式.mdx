function Demo() {
  const [val, setVal] = React.useState(0);

  return (
    <InputNumber
      label="完成度"
      min={0}
      max={1}
      step={0.01}
      formatter={(v) => `${Number(((v || 0) * 100).toFixed(2))}%`}
      parser={(v) => Number(v?.replace(/%$/, '') || '0') / 100}
      value={val}
      onChange={setVal}
      suffix={val}
    />
  );
}
