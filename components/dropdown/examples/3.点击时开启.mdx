点击时开启

```tsx
<Dropdown
  options={[
    { value: '2', label: 'option-2', options: ['A', 'B'] },
    {
      value: '1',
      label: 'option-1',
      options: [
        { value: '3', label: 'option-3', danger: true },
        { value: '4', label: 'option-4', disabled: true },
      ],
    },
  ]}
  trigger="click"
>
  <Button>点击我</Button>
</Dropdown>
```
