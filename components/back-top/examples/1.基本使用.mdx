通过 `target` 属性设置滚动容器

通过 `visibilityHeight` 属性设置容器滚动到 200 时出现返回顶部

通过 `getPopupContainer` 属性设置返回顶部按钮到插入位置

```jsx
function Demo() {
  const el = React.useRef<HTMLDivElement>(null);

  return (
    <div ref={el} style={{ height: 200, width: '100%', overflow: 'auto', position: 'relative' }}>
      <div>
        {new Array(20).fill(0).map((_, i) => (
          <p key={i}>data {i}</p>
        ))}
      </div>
      <BackTop
        visibilityHeight={200}
        target={() => el.current}
        getPopupContainer={(node) => node}
      />
    </div>
  );
}
```
