通过传入一个基础色生成出主题色调

```tsx
// import { ColorPicker, colorScheme } from 'neko-ui';

function Demo() {
  const { scheme } = colorScheme;
  const [val, setVal] = React.useState<string>('#5794ff');

  React.useEffect(() => {
    // 颜色方案发生，变更切换颜色
    injectGlobal`
    :root[data-theme=${scheme}] {
      ${generateTheme(val, { name: 'primary', dark: scheme === 'dark' })}
    }
    `;
  }, [val, scheme]);
  return <ColorPicker value={val} size="large" onChange={setVal} />;
}
```
