通过传入一个基础色生成出调色板

```jsx
// import { toneColor } from '@moneko/common';

function Demo() {
  const render = React.useCallback((colors) => {
    const ct = {
      0: colors['100'],
      1: colors['90'],
      2: colors['90'],
      3: colors['95'],
      4: colors['95'],
      5: colors['99'],
      6: colors['99'],
      7: colors['30'],
      8: colors['30'],
      9: colors['20'],
      10: colors['20'],
      11: colors['10'],
      12: colors['5'],
      13: colors['5'],
      14: colors['0'],
    };
    return (
      <div style={{ display: 'flex', gap: 8, width: '100%' }}>
        {Object.keys(colors).map((c, i) => {
          return (
            <div
              key={i}
              style={{
                flex: 1,
                paddingBlock: 4,
                textAlign: 'center',
                borderRadius: 'var(--border-radius)',
                fontSize: 8,
                fontWeight: 100,
                backgroundColor: colors[c],
                color: ct[i],
              }}
            >
              <div>{c}</div>
              <div>{colors[c]}</div>
            </div>
          );
        })}
      </div>
    );
  }, []);

  return (
    <>
      {['#fff', '#000'].map((c) => {
        const type = c === '#000';
        return (
          <div
            key={c}
            style={{
              display: 'flex',
              flexDirection: 'column',
              gap: 8,
              padding: 8,
              borderRadius: 8,
              width: '100%',
              border: 'var(--border-base)',
              backgroundColor: c,
            }}
          >
            {render(toneColor(type ? '#4d81dc' : '#5794ff', type))}
            {render(toneColor(type ? '#bb8314' : '#faad14', type))}
            {render(toneColor(type ? '#901c22' : '#ff4d4f', type))}
            {render(toneColor(type ? '#419418' : '#52c41a', type))}
          </div>
        );
      })}
    </Button>
  );
}
```
