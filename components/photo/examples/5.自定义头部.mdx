自定义头部

```tsx
function Demo() {
  const data = Array(1000)
    .fill(0)
    .map((_, i) => {
      const png = ['3.70695fb9.jpg', '4.57ff8e86.jpg', '5.7ace37c7.jpg'];

      return {
        src: `https://react-photo-view.vercel.app/_next/static/media/${png[i % 3]}`,
        alt: png[i % 3],
        key: i,
      };
    });

  return (
    <Photo
      data={data}
      header={(num) => (
        <div
          style={{
            display: 'flex',
            padding: 16,
            gap: 16,
            fontSize: 14,
            fontWeight: 'lighter',
          }}
        >
          <span>1000 / {num + 1}</span>
          <span>{data[num].alt}</span>
        </div>
      )}
    />
  );
}
```
